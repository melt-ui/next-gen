## Usage

<ThemedCode code={`
<script lang="ts">
import { TabsRoot, TabsTriggerList, TabsTrigger, TabsContent } from "melt";
</script>

<TabsRoot>
	<TabsTriggerList>
		{#each tabs as tab}
			<TabsTrigger value={tab} />
		{/each}
	</TabsTriggerList>

	{#each tabs as tab}
		<TabsContent value={tab} />
	{/each}
</TabsRoot>
`.trim()}
lang="svelte" />

<ThemedCode code={`
<script lang="ts">
	import { Tabs } from "melt";
	const tabs = new Tabs()
</script>

<div {...tabs.triggerList}>
	{#each tabIds as id}
		<button {...tabs.getTrigger(id)}>
			{id}
		</button>
	{/each}
</div>

{#each tabIds as id}
	<div {...tabs.getContent(id)}>
		{id}
	</div>
{/each}
`.trim()}
lang="svelte" />
